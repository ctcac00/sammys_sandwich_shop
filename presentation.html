-- sp_enrich_orders.sql  (118 lines)
CREATE OR REPLACE PROCEDURE sp_enrich_orders()
RETURNS VARCHAR
LANGUAGE SQL
AS $$
DECLARE
    v_rows_inserted INTEGER;
BEGIN
    TRUNCATE TABLE enriched_orders;

    INSERT INTO enriched_orders (
        order_id, customer_id, employee_id,
        location_id, order_date, order_time,
        order_type, subtotal, tax_amount,
        discount_amount, total_amount,
        payment_method, order_period,
        is_weekend, has_discount, has_tip,
        /* ... 8 more columns ... */
    )
    SELECT
        o.order_id,
        NULLIF(o.customer_id, 'NULL'),
        TRY_TO_DATE(o.order_date, 'YYYY-MM-DD'),
        TRY_TO_NUMBER(o.subtotal, 10, 2),
        CASE WHEN HOUR(...) BETWEEN 6 AND 10
             THEN 'Breakfast'
             WHEN ... THEN 'Lunch'
        END AS order_period,
        /* ... more transforms ... */
    FROM SAMMYS_RAW.orders o
    LEFT JOIN (...) ic ON o.order_id = ic.order_id
    WHERE o.order_id IS NOT NULL;

    v_rows_inserted := SQLROWCOUNT;
    RETURN 'SUCCESS: ' || v_rows_inserted;
END;
$$;

Building stg_orders — closest comparison, but key differences

Delta Live Tables

# dlt/silver/staging.py
import dlt
from pyspark.sql.functions import (
    col, trim, to_date, coalesce, lit
)
from pyspark.sql.types import DoubleType

@dlt.table(
    name="stg_orders",
    comment="Cleaned and typed order data",
    table_properties={"quality": "silver"}
)
@dlt.expect_or_drop(
    "valid_order_id",
    "order_id IS NOT NULL"
)
@dlt.expect(
    "valid_total", "total_amount >= 0"
)
def stg_orders():
    return (
        dlt.read("bronze_orders")
        .select(
            col("order_id"),
            trim(col("customer_id"))
                .alias("customer_id"),
            to_date(col("order_date"))
                .alias("order_date"),
            col("subtotal")
                .cast(DoubleType())
                .alias("subtotal"),
            # ... more columns ...
        )
    )
Python required — PySpark API, decorators, Databricks-only
